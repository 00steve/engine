50  static bool nextLineChunk(string &line,string &chunk,char &del,list<char> chars) {
51  unsigned int index = line.length();
52  if(!index) return false;
53  unsigned int i;
54  int x = chars.getCount();
55  while(x --> 0) {
56  i = line.find(chars[x]);
57  if(i < index) index = i;
58  }
59  if(index < line.length()) {
60  chunk = string(line.substr(0,index));
61  del = line[index];
62  line = string(line.substr(index+1));
63  return true;
64  }
65  chunk = line;
66  line = "";
67  return true;
68  }

75  static bool parseFile(ifstream &file,map<varMapEntry> &m,string domain) {
76  string line;
77  string key;

79  string type;
80  string chunk;
81  string last_chunk;
82  string last_value = "";

84  string last_key = "";
85  /*if a string is surrounded by " quotes, the string will be marked as
86  a literal, which means that the rest of its contents up until the next
87  " should be treated as one chunk, no matter what is in it.*/
88  bool literal = false;
89  bool was_literal = false;
90  /*the current reserved character. There aren't too many, but this
91  stores the reference to the most recent one*/
92  char rchar;
93 
94  /*when a value is stored, we first need to figure out which type it is.
95  This is the variable that will be used to do that.*/
96  char num_type;
97 
98  /*the list of characters that are used
99  to split the entire file string into different
100  chunks.*/
101  list<char> reserved_chars = list<char>();
102  reserved_chars.push(' ');
103  reserved_chars.push('[');
104  reserved_chars.push(']');
105  reserved_chars.push('#');
106 
107  getline(file, line);
108  while (!file.eof()) {
109  /*a bunch of functions to clean the string up and
110  get it ready for being parsed*/
111  line = stringTrim(line);
112  //line = convertChar(line,'{','[');
113  //line = convertChar(line,'}',']');
114  line = removeCharBefore(line,' ','[');
115  line = convertChar(line,'\t',' ');
116  line = removeRepeatingChars(line,' ');
117 
118  type = "";
119 
120  while(nextLineChunk(line,chunk,rchar,reserved_chars)) {
121  if(literal) {
122  rchar = ' ';
123  }
124  switch(rchar) {
125  case '#':
126  line = "";
127  type = "comment";
128  break;
129  case ' ':
130  if(key == "") {
131  key = chunk;
132  value = "";
133  }
134  else {
135  if(chunk[0] == '"') {
136  literal = true;
137  }
138  if(chunk[chunk.length()-1] == '"') {
139  literal = false;
140  was_literal = true;
141  }
142  value += chunk + string(1,rchar);
143  }
144 
145  type = "key/value";
146  break;
147  case '[':
148  if(chunk == "" && type == "" && last_chunk != ""){
149  chunk = last_chunk;
150  }
151  //cout << " - [object] - " << chunk << endl;
152  type = "object";
153  line = stringTrim(line);
154  parseFile(file,m,domain+chunk+".");
155  break;
156  case ']':
157  //cout << " - [end object]\n";
158  type = "endobject";
159  return true;
160  break;
161  }
162 
163  if(!literal && last_key == key && last_key != "" && last_type == "key/value") {
164  value = stringTrim(value);
165  if(was_literal) {
166  was_literal = false;
167  value = value.substr(1,value.length()-2);
168  }
169  /*based on the way that all of this shit works, there is an
170  extra char at the end of every value that is to be saved. It
171  needs to be removed. If it isn't, any strings will have an
172  extra space or " at the end and numbers won't be evaluated as
173  numbers.*/
174  value = value.substr(0,value.length()-1);
175 
176  //cout << "SAVE: " << domain << last_key << " = " << value << endl;
177  key = domain + key;
178  if(stringIsNumber(value,num_type)) {
179  num_type = 'D';
180  switch(num_type) {
181  case 'U': push<unsigned int>(m,new unsigned int(stringToNumber<unsigned int>(value)),key); break;
182  case 'I': push<int>(m,new int(stringToNumber<int>(value)),key); break;
183  case 'F': push<float>(m,new float(stringToNumber<float>(value)), key); break;
184  case 'D': push<double>(m,new double(stringToNumber<double>(value)), key); break;
185  }
186  //cout << " - NUMBER (" << num_type << ")\n";
187  }
188  else if(value == "true"){
189  push<bool>(m,new bool(true),key);
190  }else if(value == "false"){
191  push<bool>(m,new bool(false),key);
192  }
193  else {
194  push<string>(m,new string(value),key);
195  //cout << " - STRING\n";
196  }
197  key = "";
198  value = "";
199  }
200  last_key = key;
201  last_value = value;
202  last_chunk = chunk;
203  last_type = type;
204  }//end of looping through chunks in line
205 
206  getline(file, line);
207  }
208  return true;
209  }
