87  static void loadAll(string fileName){
88  list<dependency*> dependencies = list<dependency*>();
89  dependency *cD = new dependency();
90  dependency *cC = new dependency();
91  loader* fLoader = NULL;
92  cD->object_reference = NULL;
93  cD->file = fileName;
94  dependencies.push(cD);
95  while(dependencies.getCount()){
96  cD = dependencies.pop();
97  if(!directory.exists(cD->file)){
98  fLoader = getFileLoader(cD->file);
99  if(!fLoader) continue;
100  fLoader->nonThreadedStart();
101  if(!fLoader->wasSuccessful()) continue;
102  cD->load_data = fLoader->getData();
103  /*loop through all of the dependencies that have
104  been found for the file. Check to see if the file
105  has been loaded, if not, add it to the list of
106  dependencies to load*/
107  while(fLoader->getDependencies()->getCount()){
108  cC = fLoader->getDependencies()->pop();
109  dependencies.push(cC);
110  }
111  }//end of if data needs to be loaded
112  if(cD->object_reference){
113  ((loader*)cD->object_reference)->integrateDependency(cD);
114  }
115 
116  }//end of looping through dependencies
117  }
