65  virtual void recalculateBounds() {
66  #if EVENT_VERBOSE_OUTPUT
67  cout << "RECALCULATE view " << this->getId() << "\n";
68  #endif
69  switch(size_method){
70  case VIEW_SIZEBY_PIXEL:
71  calculated_offset = offset;
72  break;
73  case VIEW_SIZEBY_SCALE:
74  dimensions = int2(windowSize.x*scale.x,windowSize.y*scale.y);
75  raw_dimensions = double2((float)windowSize.x*scale.x,(float)windowSize.y*scale.y);
76  calculated_offset = double2(windowSize.x*offset.x,windowSize.y*offset.y);
77  break;
78  }
79  //calculate aspect ratio
80  aspectRatio = (float)dimensions.x / (float)dimensions.y;
81  //find the bounds of the view based on its location setting
82  switch(location) {
83  case VIEW_LOCATION_TOP_LEFT:
84  bottom_left = int2(windowSize.x*calculated_offset.x,windowSize.y-dimensions.y-windowSize.y*calculated_offset.y);
85  break;
86  case VIEW_LOCATION_TOP:
87  bottom_left = int2((windowSize.x - dimensions.x) * .5f +windowSize.x*calculated_offset.x , windowSize.y-dimensions.y-windowSize.y*calculated_offset.y);
88  break;
89  case VIEW_LOCATION_TOP_RIGHT:
90  bottom_left = int2(windowSize.x-dimensions.x-windowSize.x*calculated_offset.x,windowSize.y-dimensions.y-windowSize.y*calculated_offset.y);
91  break;
92  case VIEW_LOCATION_LEFT:
93  bottom_left = int2( windowSize.x*calculated_offset.x, (windowSize.y - dimensions.y) * .5f);
94  break;
95  case VIEW_LOCATION_CENTER:
96  bottom_left = int2((windowSize.x - dimensions.x) * .5f +windowSize.x*calculated_offset.x , (windowSize.y - dimensions.y) * .5f);
97  break;
98  case VIEW_LOCATION_RIGHT:
99  bottom_left = int2( windowSize.x-dimensions.x-windowSize.x*calculated_offset.x, (windowSize.y - dimensions.y) * .5f);
100  break;
101  case VIEW_LOCATION_BOTTOM_LEFT:
102  bottom_left = int2( windowSize.x*calculated_offset.x, windowSize.y*calculated_offset.y);
103  break;
104  case VIEW_LOCATION_BOTTOM:
105  bottom_left = int2((windowSize.x - dimensions.x) * .5f , windowSize.y*calculated_offset.y);
106  break;
107  case VIEW_LOCATION_BOTTOM_RIGHT:
108  bottom_left = int2(windowSize.x-dimensions.x-windowSize.x*calculated_offset.x, windowSize.y*calculated_offset.y);
109  break;
110  }
111  }
