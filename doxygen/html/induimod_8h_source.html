168  virtual void update(){
169  switch(state){
170  //if the mouse is not even over the module
171  case 0:
172  //check for mouse going over the drag zone (and not clicking) or over any sub modules
173  if(*mX > offsetX && *mX< offsetX + width && *mY < offsetY && !(*mC)[0]) {
174  if(*mY > offsetY - dragAreaHeight) {
175  state = 1;
176  cout << "state = 1 entered drag area\n";
177  }
178  else if(*mY > offsetY - height) {
179  state = 10;
180  cout << "state = 10 entered widget area\n";
181  }
182  }
183  break;
184  //if the mouse is over the drag zone
185  case 1:
186  //check to see if the mouse left the drag zone.
187  if(offsetX > *mX || offsetX + width < *mX || offsetY < *mY) {
188  state = 0;
189  cout << "state = 0 left module area\n";
190  break;
191  }
192  //check to see if state went over to the widget portion of the module
193  if(offsetY - dragAreaHeight > *mY) {
194  cout << "state = 10 entered widget area\n";
195  state = 10;
196  }
197  //check to see if the mouse has been clicked. If it is, set dragging to true
198  if((*mC)[0]) {
199  state = 5;
200  cout << "state = 5 start dragging\n";
201  }
202 
203  break;
204  //setup variables needed for when the module is being dragged
205  case 5:
206  cursorOffsetX = *mX-offsetX;
207  cursorOffsetY = *mY-offsetY;
208  state = 6;
209  //exclue the break so that it runs the case 6 code right away
210  //if the module is being dragged. Check for the mouse unclicking
211  case 6:
212  offsetX = *mX - cursorOffsetX;
213  offsetY = *mY - cursorOffsetY;
214  //check to make sure that the module is not going out of the screen
215  if(offsetX < 0) offsetX = 0;
216  else if(offsetX+width > *screenSizeX) offsetX = *screenSizeX - width;
217  if(offsetY - height < 0) offsetY = height;//*screenSizeY +
218  else if(offsetY > *screenSizeY) offsetY = *screenSizeY;
219 
220  //check to see if the mouse is still being clicked
221  if(!(*mC)[0]) {
222  if(*mX > offsetX && *mX< offsetX + width && *mY < offsetY) {
223  if(*mY > offsetY - dragAreaHeight) {
224  state = 1;
225  cout << "state = 1 entered drag area\n";
226  break;
227  }
228  else if(*mY > offsetY - height) {
229  state = 10;
230  cout << "state = 10 entered widget area\n";
231  break;
232  }
233  }
234  //if mouse is not over the drag area or widget area, default to
235  //being out of box
236  cout << "state = 0 out of module\n";
237  state = 0;
238  }//end of if mouse is not being clicked any more
239  break;
240 
241  //if the cursor is over the widget part of the module
242  case 10:
243  if(currentWidget > -1) {
244  //update the widget until it returns false
245  if(!widgets[currentWidget]->update(*mX-offsetX,*mY-boff,mC,kb) && !widgets[currentWidget]->Active()) {
246  currentWidget = -1;
247  }
248  }
249  else {
250  //figure out which widget should be updated based on the
251  //mouse position
252  toff = offsetY-dragAreaHeight;
253  for(int i=0;i<widgetCount;i++) {
254  //figure out where the bottom of the current widget is
255  boff = toff-widgets[i]->Height();
256  //check if the mouse is within this widget
257  if(*mY > boff && *mY < toff) {
258  widgets[i]->update(*mX-offsetX,*mY-boff,mC,kb);
259  currentWidget = i;
260  break;
261  }
262  //set the top offset to what the bottom of the last widget was
263  toff = boff;
264  }

267  //has left the area of the modules.
268  if(!(*mC)[0]) {
269  //check to see if the mouse left the widget zone

271  state = 0;
