24  void custom_load(string fileName){
25  #if DEBUG_LOAD
26  cout << "load::img(" << fileName << ")\n";
27  #endif
28  string ext;
29  getExt(fileName,ext);
30  FREE_IMAGE_FORMAT extType;
31  //try to load the image into the program
32  ext = stringToLower(ext);
33  if(ext == "png") {
34  extType = FIF_PNG;
35  } else if(ext == "bmp"){
36  extType = FIF_BMP;
37  } else if(ext == "jpg"){
38  extType = FIF_JPEG;
39  } else {
40  failed();
41  return;
42  }
43  FIBITMAP *bitmap = FreeImage_Load(extType, fileName.c_str(), 0);
44  //make sure that the image was loaded successfully
45  if (bitmap) {
46  // bitmap successfully loaded!
47  int w = FreeImage_GetWidth(bitmap);
48  int h = FreeImage_GetHeight(bitmap);
49  #if DEBUG_IMAGE_LOAD
50  cout << "width: " << w << endl;
51  cout << "height: " << h << endl;
52  #endif
53  GLuint id;
54  GLubyte* textura = new GLubyte[4*w*h];
55  //FreeImage loads in BGR format, so you need to swap some bytes(Or use GL_BGR).
56  int bpp = FreeImage_GetBPP(bitmap);
57  if(bpp == 24){
58  #if DEBUG_IMAGE_LOAD
59  cout << "convert image from 24 to 32 bit.\n";
60  #endif
61  bitmap = FreeImage_ConvertTo32Bits(bitmap);
62  bpp = FreeImage_GetBPP(bitmap);
63  }
64  char* pixels = (char*)FreeImage_GetBits(bitmap);
65  int s = w*h*4;
66  for(int j= 0; j<s; j+=4){
67  textura[j+0]= pixels[j+2];
68  textura[j+1]= pixels[j+1];
69  textura[j+2]= pixels[j+0];
70  textura[j+3]= pixels[j+3];
71  }
72  glGenTextures(1, &id);
73  glBindTexture(GL_TEXTURE_2D, id);
74  #if DEBUG_IMAGE_LOAD
75  cout << "bound texture.\n";
76  #endif
77 
78  if(bpp == 24) {
79  glTexImage2D(GL_TEXTURE_2D,0,GL_RGBA, w, h, 0, GL_RGB,GL_UNSIGNED_BYTE,(GLvoid*)textura );
80  } else if(bpp == 32) {
81  glTexImage2D(GL_TEXTURE_2D,0,GL_RGBA, w, h, 0, GL_RGBA,GL_UNSIGNED_BYTE,(GLvoid*)textura );
82  }
83  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
84  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
85  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
86  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
87  glGenerateMipmap(GL_TEXTURE_2D);
88  FreeImage_Unload(bitmap);//clear the png from memory
89  //store the name of the file that was loaded
90  tex = new texture(id,w,h);
91  } else {
92  failed();
93  return;
94  }
95 
96 
97  //cout << "finished loading " << fileName << "\n";
98  succeeded();
99  return;
100  }
